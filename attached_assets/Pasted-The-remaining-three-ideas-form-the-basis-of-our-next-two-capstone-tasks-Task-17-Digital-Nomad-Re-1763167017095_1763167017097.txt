// server/routes.ts (ADD THIS ROUTE)

// Digital Nomad Readiness Score API endpoint (Task #17)
app.post('/api/wifi_readiness', async (req, res) => {
    try {
        const { city } = req.body;

        if (!city) {
            return res.status(400).json({ error: "Missing city parameter." });
        }

        const apiKey = process.env.GROQ_API_KEY;
        if (!apiKey) {
            return res.status(503).json({ error: "Groq API key not found." });
        }

        const groq = new Groq({ apiKey });

        const systemPrompt = (
            "You are a remote work consultant and digital nomad expert. " +
            "Analyze the specified city for its digital infrastructure. " +
            "Based on the city's known average broadband speed, power reliability, and co-working/cafe culture, " +
            "generate a 'Digital Nomad Readiness Score' out of 100. " +
            "Output ONLY a JSON object with three fields: " +
            "'readinessScore' (An integer from 60 to 99), " +
            "'internetSummary' (A 3-sentence justification focusing on typical speed, reliability, and best local work areas), and " +
            "'bestProviderTip' (A concise tip on the most reliable internet provider in that city)."
        );

        const completion = await groq.chat.completions.create({
            model: "llama-3.3-70b-versatile",
            messages: [
                { role: "system", content: systemPrompt },
                { role: "user", content: `Generate the Digital Nomad Readiness Score for ${city}.` }
            ],
            response_format: { type: "json_object" }
        });

        const jsonResponseText = completion.choices[0].message.content?.trim() || '{"readinessScore": 75, "internetSummary": "Data inconclusive.", "bestProviderTip": "Ask a local expat forum."}';
        const nomadData = JSON.parse(jsonResponseText);

        return res.json(nomadData);

    } catch (error: any) {
        console.error('Digital Nomad API Error:', error);
        return res.status(500).json({ error: 'AI digital nomad service failed to process request.' });
    }
});

const shareScreenReport = () => {
    if (navigator.share) {
        // Use native share API for modern browsers/mobile
        navigator.share({
            title: `Relocation Report: ${neighborhood.name}, ${neighborhood.city}`,
            text: "See my personalized report. Let's screen share this later!",
            url: window.location.href

        navigator.clipboard.writeText(window.location.href);

};
