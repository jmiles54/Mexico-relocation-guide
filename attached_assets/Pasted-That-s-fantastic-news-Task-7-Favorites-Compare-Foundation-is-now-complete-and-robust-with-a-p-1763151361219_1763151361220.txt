// client/src/pages/Neighborhood.tsx (Inside the Neighborhood component function)

// Destructure the city data needed
const cityId = neighborhood.cityId as 'pv' | 'merida' | 'sma' | 'cancun'; 
const cityName = neighborhood.city;

// The function to call the Groq/Flask backend
const getSeniorScore = async () => {
    // 1. Target the correct output element
    const resultDiv = document.getElementById(`${cityId}-senior-result`);
    if (!resultDiv) return;

    // Set loading state for instant feedback
    resultDiv.innerHTML = '<strong class="text-yellow-500">Calculating... (Powered by Groq LPU)</strong>';

    try {
        // 2. Call your Flask/Groq backend endpoint
        const response = await fetch('/api/senior_score', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            // Send the full city name to the Python endpoint
            body: JSON.stringify({ city: cityName }) 
        });

        const data = await response.json();

        // 3. Handle errors from the server or Groq API
        if (!response.ok || data.error) {
            resultDiv.innerHTML = `<span class="text-red-500">Error: ${data.error || 'Failed to fetch score.'}</span>`;
            console.error('API Error:', data.error);
            return;
        }

        // 4. Display the instant score and justification
        resultDiv.innerHTML = `
            <div class="text-5xl font-extrabold text-green-400">
                ${data.score}% Fit
            </div>
            <p class="text-sm text-muted-foreground mt-2">
                ${data.justification}
            </p>
        `;

    } catch (error) {
        console.error('Fetch error:', error);
        resultDiv.innerHTML = `<span class="text-red-500">Connection Error. Is main.py running?</span>`;
    }
};
