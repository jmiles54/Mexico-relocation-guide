sql-- Enable PostGIS for geographic data
CREATE EXTENSION IF NOT EXISTS postgis;

-- ============================================
-- CORE LOCATION TABLES
-- ============================================

-- Countries table (for future expansion)
CREATE TABLE countries (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    code CHAR(2) NOT NULL UNIQUE, -- ISO 3166-1 alpha-2
    currency_code CHAR(3) NOT NULL,
    currency_symbol VARCHAR(5),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Cities table
CREATE TABLE cities (
    id SERIAL PRIMARY KEY,
    country_id INTEGER REFERENCES countries(id),
    name VARCHAR(100) NOT NULL,
    state_province VARCHAR(100),
    population INTEGER,
    timezone VARCHAR(50),
    coordinates GEOGRAPHY(POINT, 4326), -- PostGIS geographic point
    elevation_meters INTEGER,
    climate_zone VARCHAR(50),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Neighborhoods table (core of the app)
CREATE TABLE neighborhoods (
    id SERIAL PRIMARY KEY,
    city_id INTEGER REFERENCES cities(id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    slug VARCHAR(100) NOT NULL, -- URL-friendly version
    description TEXT,
    location_description TEXT, -- "South of Río Cuale, beachfront"
    
    -- Geographic data
    coordinates GEOGRAPHY(POINT, 4326), -- Center point
    boundary GEOGRAPHY(POLYGON, 4326), -- Neighborhood boundary if available
    
    -- Meta
    is_active BOOLEAN DEFAULT TRUE,
    display_order INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    
    UNIQUE(city_id, slug)
);

-- Create spatial indexes for performance
CREATE INDEX idx_cities_coordinates ON cities USING GIST(coordinates);
CREATE INDEX idx_neighborhoods_coordinates ON neighborhoods USING GIST(coordinates);
CREATE INDEX idx_neighborhoods_boundary ON neighborhoods USING GIST(boundary);

-- ============================================
-- AFFORDABILITY DATA
-- ============================================

-- Housing prices with full transparency
CREATE TABLE housing_prices (
    id SERIAL PRIMARY KEY,
    neighborhood_id INTEGER REFERENCES neighborhoods(id) ON DELETE CASCADE,
    
    -- What type of housing
    housing_type VARCHAR(50) NOT NULL, -- '1BR_furnished', '2BR_furnished', '1BR_unfurnished', etc.
    
    -- The actual data
    price_amount DECIMAL(10,2) NOT NULL,
    currency_code CHAR(3) NOT NULL,
    price_period VARCHAR(20) DEFAULT 'monthly', -- 'monthly', 'weekly', 'nightly'
    
    -- Location specificity
    location_detail VARCHAR(100), -- 'downtown', 'outskirts', 'beachfront'
    
    -- Data provenance (CRITICAL for transparency)
    data_source VARCHAR(100) NOT NULL, -- 'Facebook Marketplace', 'Inmuebles24', 'User submission'
    source_url TEXT, -- Link to actual listing if available
    source_detail TEXT, -- "avg of 12 listings", "single listing"
    collected_date DATE NOT NULL,
    verified_by_users INTEGER DEFAULT 0, -- How many users confirmed this

-- View for latest housing prices by neighborhood
CREATE VIEW latest_housing_prices AS
SELECT DISTINCT ON (neighborhood_id, housing_type, location_detail)
    neighborhood_id,
    housing_type,
    location_detail,
    price_amount,
    currency_code,
    data_source,
    collected_date,
    verified_by_users
FROM housing_prices
WHERE is_outlier = FALSE
  AND admin_verified = TRUE
ORDER BY neighborhood_id, housing_type, location_detail, collected_date DESC;

-- View for current affordability scores
CREATE VIEW current_affordability AS
SELECT DISTINCT ON (neighborhood_id)
    neighborhood_id,
    affordability_score,
    comparison_text,
    calculation_date,
    data_points_count
FROM affordability_scores
ORDER BY neighborhood_id, calculation_date DESC;

-- View for neighborhood summary (denormalized for performance)
CREATE VIEW neighborhood_summary AS
SELECT 
    n.id,
    n.name,
    n.slug,
    c.name AS city_name,
    n.description,
    
    -- Latest affordability
    a.affordability_score,
    
    -- Latest safety
    s.night_safety_score,
    
    -- Counts
    (SELECT COUNT(*) FROM social_venues WHERE neighborhood_id = n.id AND is_active = TRUE) AS venue_count,
    (SELECT COUNT(*) FROM restaurants WHERE neighborhood_id = n.id AND is_active = TRUE) AS restaurant_count,
    (SELECT COUNT(*) FROM venue_events ve JOIN social_venues sv ON ve.venue_id = sv.id 
     WHERE sv.neighborhood_id = n.id AND ve.is_active = TRUE) AS weekly_events,
    
    -- Average user rating
    (SELECT AVG(rating) FROM neighborhood_reviews WHERE neighborhood_id = n.id) AS avg_user_rating,
    (SELECT COUNT(*) FROM neighborhood_reviews WHERE neighborhood_id = n.id) AS review_count,
    
    n.coordinates
FROM neighborhoods n
JOIN cities c ON n.city_id = c.id
LEFT JOIN current_affordability a ON n.id = a.neighborhood_id
LEFT JOIN LATERAL (
    SELECT night_safety_score
    FROM safety_scores
    WHERE neighborhood_id = n.id
    ORDER BY period_end DESC
    LIMIT 1
) s ON true
WHERE n.is_active = TRUE;

Sample Queries
sql-- Get all neighborhoods in Puerto Vallarta with their latest data
SELECT 
    ns.*,
    hp.price_amount AS rent_1br,
    hp.data_source AS rent_source
FROM neighborhood_summary ns
LEFT JOIN latest_housing_prices hp ON ns.id = hp.neighborhood_id 
    AND hp.housing_type = '1BR_furnished' 
    AND hp.location_detail = 'downtown'
WHERE ns.city_name = 'Puerto Vallarta'
ORDER BY ns.affordability_score DESC;

-- Find neighborhoods within walking distance (800m) of a specific beach
SELECT 
    n.name,
    np.distance_meters,
    np.walk_time_minutes,
    poi.name AS beach_name
FROM neighborhoods n
JOIN neighborhood_proximity np ON n.id = np.neighborhood_id
JOIN points_of_interest poi ON np.poi_id = poi.id
WHERE poi.poi_type = 'beach'
  AND np.distance_meters <= 800
ORDER BY np.distance_meters;

## Critical Insights

### 1. **APIs Are NOT Your Biggest Cost**
At scale, your biggest costs are:
- Human labor (support, moderation, content)
- Marketing/customer acquisition
- Hosting/infrastructure

APIs will be 10-20% of total costs.

### 2. **Free Tier is Sufficient Until 10k MAU**
Don't prematurely optimize. The free stack can support:
- 10,000 monthly active users
- Millions of data points
- Excellent user experience

### 3. **Crowdsourcing > APIs for Unique Data**
The most valuable data **can't** be API'd:
- "Which wine bar has the best 50+ women's nights?"
- "Is this restaurant actually frequented by locals?"
- "Does this neighborhood feel safe at night?"

APIs give you breadth. Community gives you depth.

### 4. **Manual Curation Has Better ROI Than APIs (Early On)**

**Example: Restaurant pricing**

Option A: Google Places API
- Cost: $17 per 1,000 requests
- Data quality: Generic, not local-focused

Option B: VA Manual Collection
- Cost: $50 for 10 hours
- Can research: 50 restaurants with photos, menus, local insights
- Data quality: Superior

**Winner in early stages: Manual curation**

### 5. **User Contributions Scale Better Than APIs**

1,000 engaged users each contributing 1 data point = 1,000 fresh data points/month for $0.

Compare to:
- Paying for API calls: $100-500/month
- Hiring VAs: $200-400/month
- Manual work: 40 hours/month

**Conclusion:** Build contribution features BEFORE paying for APIs.
